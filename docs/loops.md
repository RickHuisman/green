# Loops

## For
for x in 0..3 {
    print(x); // 0 1 2 3
}

for x in 1 to 4 {
    print(x); // 1 2 3 4
}

for x in 4 downto 1 step 2 {
    print(x); // Prints 4-2
}

for item in collection {
    print(item);
}

## While
var x = 0;
while x < 5 {
    x = x++
    print(x) // 1 2 3 4 5
}
